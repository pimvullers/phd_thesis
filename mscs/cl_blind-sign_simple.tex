\begin{msc}{}

\fontfamily{\rmdefault}

\setlength{\instwidth}{1.5\instwidth}
\setlength{\instdist}{14cm}
\setlength{\envinstdist}{2\envinstdist}
\setlength{\referenceoverlap}{6cm}

\declinst{R}{}{Receiver}
\declinst{S}{}{Signer}

\action*{%
  random $v'$\\ 
  $U \leftarrow S^{v'} \prod_{i \in I_s} R_i^{m_i} \mod n$%
}{R}
\nextlevel[3.2]

\mess{$U$}{R}{S}
\nextlevel[.5]

\referencestart{PK}{%
  $PK\{(\nu', \{\mu_i : i \in I_s\}) : U \equiv S^{\nu'} \prod_{i \in I_s} R_i^{\mu_i} \mod n \}$%
}{R}{S}
\nextlevel[1.8]
\referenceend{PK}
\nextlevel[.5]

\action*{%
  random prime $e$\\
  random $v''$\\
  $A \leftarrow \left(\dfrac{Z}{U S^{v''}\prod_{i \in I_p} R_i^{m_i}} \right)^{1/e} \mod n$%
}{S}
\nextlevel[4.8]

\mess{$(e, v'', A)$}{S}{R}
\nextlevel[.5]

\referencestart{PK2}{%
  $PK\{(\delta) : A \equiv \left(\dfrac{Z}{U S^{v''} \prod_{i \in I_p} R_i^{m_i}} \right)^{\delta} \mod n \}$%
}{R}{S}
\nextlevel[2.8]
\referenceend{PK2}
\nextlevel[.5]

\action*{%
  verify $e$ is prime\\
  $v \leftarrow v' + v''$%
}{R}
\nextlevel[2]

\end{msc}
